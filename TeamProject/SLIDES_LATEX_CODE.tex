%% --- SLIDE 1: The Problem ---

\begin{frame}{The Problem: Predicting Ocean \pce{CO2}}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Why this matters:}
        \begin{itemize}
            \item The ocean absorbs $\sim$25\% of human \pce{CO2} emissions
            \item Rising ocean \pce{CO2} drives acidification, threatening marine ecosystems
            \item We need global p\pce{CO2} maps to track carbon uptake --- but buoys only cover a few points
        \end{itemize}

        \column{0.5\textwidth}
        \begin{itemize}
            \item \textbf{Goal:} Predict ocean surface p\pce{CO2} using satellite SST
            \item Buoys measure p\pce{CO2} directly --- but only at discrete locations
            \item Satellites measure SST everywhere --- but can't see \pce{CO2}
            \item \textbf{Idea:} Learn the SST $\rightarrow$ p\pce{CO2} relationship at buoy sites, then predict p\pce{CO2} anywhere
        \end{itemize}
    \end{columns}
\end{frame}

%% --- SLIDE 4: Spatial Matching Discussion ---

\begin{frame}{Spatial Matching: How Big Is Our Bounding Box?}

    %% Comparison table at top
    \begin{center}
    \small
    \begin{tabular}{l l l}
        \textbf{Analysis Type} & \textbf{Typical Box Size} & \textbf{Use Case} \\
        \hline
        Satellite--buoy validation & $\sim$25 km radius & Point matching \\
        Regional coastal studies & $0.1^\circ$--$0.25^\circ$ ($\sim$10--25 km) & Shelf/upwelling zones \\
        Global pCO2 products (SOCAT) & $1^\circ \times 1^\circ$ ($\sim$110 km) & Monthly gridded maps \\
        \textbf{Our approach} & $\mathbf{2^\circ \times 2^\circ}$ \textbf{($\sim$220 km)} & \textbf{Exploratory first pass} \\
    \end{tabular}
    \end{center}

    \vspace{0.3cm}
    \hrule
    \vspace{0.3cm}

    \small
    \textbf{Our situation:} Most buoys are near coastlines --- a 220 km box can span across shore, upwelling zones, or different water masses, potentially diluting the local SST signal.

    \vspace{0.2cm}
    \textbf{Next steps:}
    \begin{itemize}
        \item Test smaller boxes ($\pm 0.25^\circ$ or $\pm 0.1^\circ$) and compare model accuracy
        \item Use different box sizes for coastal vs.\ open-ocean sites
    \end{itemize}

\end{frame}
